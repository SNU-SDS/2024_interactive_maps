1.3 샘플 데이터

2022년 기준 서울시 구간 인구이동 데이터를 활용한다.

df_2022 <- read_excel("D:/My R/Korea Census Data Manipulation/9 Migration/DM_Comp_Code_Mig_2022_Sidogungu.xlsx") 

df <- df_2022 |> 
  rename_with(
    ~ str_replace(.x, c("Total_T", "Total_M", "Total_F"), c("TTotal", "MTotal", "FTotal")),
    contains(c("Total_T", "Total_M", "Total_F"))
  ) |> 
  filter(
    (Ori_CD > 10000 & Ori_CD < 20000) & (Des_CD > 10000 & Des_CD < 20000) & (Ori_CD != Des_CD)
  )

데이터를 정리하여 룩업용으로 사용한다.

out_mig_sum <- df |> 
  summarize(
    SGG1_CD = first(Ori_CD), 
    SGG1_NM = first(Ori_NM),
    out_mig = sum(TTotal),
    .by = Ori_CD
  ) 
in_mig_sum <- df |> 
  summarize(
    SGG1_CD = first(Des_CD), 
    SGG1_NM = first(Des_NM),
    in_mig = sum(TTotal),
    .by = Des_CD
  ) 
mig_sum <- out_mig_sum |> 
  left_join(
    in_mig_sum |> select(SGG1_CD, in_mig), 
    by = join_by(SGG1_CD)
  ) |> 
  mutate(
    gr_mig = out_mig + in_mig, 
    nt_mig = in_mig - out_mig
  )

2.2 Mapbox 토큰 등록하기

Mapbox에 가서 무료 토큰을 생성한다. 그리고 나서 Sys.setenv() 함수를 활용하여 등록한다. 다음의 형식을 취한다.

Sys.setenv(MAPBOX_API_TOKEN = "YOUR_MAPBOX_ACCESS_TOKEN")

2.3 위치와 플로 데이터의 설정

위치 데이터는 서울시 구의 센트로이드에 대한 좌표값이 포함된 데이터를 의미한다.

coords <- seoul_gu |> 
  st_transform(crs = 4326) |> 
  st_centroid() |> 
  st_coordinates()
locations <- seoul_gu |> 
  mutate(
    lon = coords[, 1],
    lat = coords[, 2],
    id = SGG1_CD, 
    name = SGG1_NM
  ) |> 
  st_drop_geometry() |> 
  select(
    id, name, lon, lat
  )

플로 데이터는 방향적 지역쌍별 플로량을 포함하고 있는 데이터이다.

flows <- df |> 
  select(
    origin = Ori_CD,
    dest = Des_CD,
    count = TTotal
  )

지도를 제작한다.

flowmapblue <- flowmapblue(
  locations = locations,
  flows = flows,
  mapboxAccessToken = Sys.getenv('MAPBOX_API_TOKEN'),
  clustering = TRUE,
  darkMode = TRUE,
  animation = FALSE
)

지도를 저장한다.

library(htmlwidgets)
saveWidget(flowmapblue, file = "flowmapblue.html")

애니메이션 지도를 제작해 본다.

flowmapblue_ani <- flowmapblue(
  locations = locations,
  flows = flows,
  mapboxAccessToken = Sys.getenv('MAPBOX_API_TOKEN'),
  clustering = TRUE,
  darkMode = TRUE,
  animation = TRUE
)
saveWidget(flowmapblue_ani, file = "flowmapblue_ani.html")
